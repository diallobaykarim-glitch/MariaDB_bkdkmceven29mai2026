# 🧠 ÉTAPE 1 — AJOUTER UN PATIENT (POST)

Ajoute dans `main.py` :

```python id="post1"
from fastapi import FastAPI
from pydantic import BaseModel
import mariadb

app = FastAPI(title="ORL IA API")

class Patient(BaseModel):
    nom: str
    age: int
    diagnostic: str

def get_connection():
    return mariadb.connect(
        host="localhost",
        user="root",
        password="root123",
        database="orl_ia"
    )

@app.post("/patients")
def add_patient(p: Patient):
    conn = get_connection()
    cur = conn.cursor()

    cur.execute(
        "INSERT INTO patients (nom, age, diagnostic) VALUES (?, ?, ?)",
        (p.nom, p.age, p.diagnostic)
    )

    conn.commit()
    conn.close()

    return {"message": "Patient ajouté ✔"}
```
