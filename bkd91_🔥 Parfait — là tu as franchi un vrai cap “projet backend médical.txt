@app.get("/patients")
def get_patients():
    conn = get_connection()
    cur = conn.cursor()
    cur.execute("SELECT * FROM patients")
    rows = cur.fetchall()
    conn.close()

    return [
        {
            "id": r[0],
            "nom": r[1],
            "age": r[2],
            "diagnostic": r[3]
        }
        for r in rows
    ]
