```python id="mainapi1"
from fastapi import FastAPI
from pydantic import BaseModel
import mariadb
import pandas as pd

from sklearn.preprocessing import LabelEncoder
from sklearn.tree import DecisionTreeClassifier

app = FastAPI(title="ORL IA API")

# =========================
# CONNEXION DB
# =========================
def get_connection():
    return mariadb.connect(
        host="localhost",
        user="root",
        password="root123",
        database="orl_ia"
    )

# =========================
# CHARGEMENT DATA + TRAIN MODEL
# =========================
def train_model():
    conn = get_connection()
    cur = conn.cursor()

    cur.execute("""
        SELECT age, nom, diagnostic,
               obstruction_nasale,
               rhinorrhee,
               epistaxis,
               anosmie,
               tabac
        FROM patients
    """)

    rows = cur.fetchall()
    conn.close()

    df = pd.DataFrame(
        rows,
        columns=[
            "age", "nom", "diagnostic",
            "obstruction_nasale",
            "rhinorrhee",
            "epistaxis",
            "anosmie",
            "tabac"
        ]
    )

    # sexe
    df["sexe"] = df["nom"].apply(lambda x: 1 if "-M" in x else 0)

    X = df[
        [
            "age", "sexe",
            "obstruction_nasale",
            "rhinorrhee",
            "epistaxis",
            "anosmie",
            "tabac"
        ]
    ]

    encoder = LabelEncoder()
    y = encoder.fit_transform(df["diagnostic"])

    model = DecisionTreeClassifier(max_depth=6)
    model.fit(X, y)

    return model, encoder


model, encoder = train_model()

# =========================
# MODELE API INPUT
# =========================
class PatientInput(BaseModel):
    age: int
    sexe: int
    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int

# =========================
# ROUTE PATIENTS
# =========================
@app.get("/patients")
def get_patients():
    conn = get_connection()
    cur = conn.cursor()

    cur.execute("SELECT * FROM patients")
    rows = cur.fetchall()

    conn.close()

    return rows

# =========================
# ROUTE PREDICTION IA
# =========================
@app.post("/predict")
def predict(p: PatientInput):

    sample = pd.DataFrame(
        [[
            p.age,
            p.sexe,
            p.obstruction_nasale,
            p.rhinorrhee,
            p.epistaxis,
            p.anosmie,
            p.tabac
        ]],
        columns=[
            "age", "sexe",
            "obstruction_nasale",
            "rhinorrhee",
            "epistaxis",
            "anosmie",
            "tabac"
        ]
    )

    pred = model.predict(sample)
    result = encoder.inverse_transform(pred)

    return {
        "prediction": result[0]
    }
```

import mariadb
import pandas as pd
import joblib

from sklearn.preprocessing import LabelEncoder

# =========================
# DB CONNECTION
# =========================
conn = mariadb.connect(
    host="localhost",
    user="root",
    password="root123",
    database="orl_ia"
)

cur = conn.cursor()

cur.execute("""
    SELECT age, nom, diagnostic,
           obstruction_nasale,
           rhinorrhee,
           epistaxis,
           anosmie,
           tabac
    FROM patients
""")

rows = cur.fetchall()
conn.close()

df = pd.DataFrame(rows, columns=[
    "age", "nom", "diagnostic",
    "obstruction_nasale",
    "rhinorrhee",
    "epistaxis",
    "anosmie",
    "tabac"
])

# sexe
