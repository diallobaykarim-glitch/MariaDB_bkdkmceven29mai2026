```python id="r0zfdw"
import random
import mariadb

conn = mariadb.connect(
    host="localhost",
    user="root",
    password="root123",
    database="orl_ia"
)

cur = conn.cursor()

# =========================
# RESET TABLE
# =========================
cur.execute("DELETE FROM patients")

# =========================
# PATHOLOGIES
# =========================
pathologies = (
    ["Cancer du cavum"] * 7 +
    ["Cancer sinus maxillaire"] * 5 +
    ["Sinusite"] * 53 +
    ["Allergie naso-sinusienne"] * 55 +
    ["Autre pathologie rhinologique"] * 9
)

# =========================
# NOMS
# =========================
noms_h = [
    "Diallo", "Sow", "Ndiaye", "Ba", "Fall",
    "Diop", "Sarr", "Kane", "Ndao", "Gueye"
]

noms_f = [
    "Fatou", "Awa", "Mariama", "Khady", "Aminata",
    "Rokhaya", "Adama", "Yacine", "Coumba", "Astou"
]

# =========================
# INSERTION
# =========================
for diag in pathologies:

    sexe = random.choice(["M", "F"])

    if sexe == "M":
        nom = random.choice(noms_h)
    else:
        nom = random.choice(noms_f)

    age = random.randint(5, 85)

    cur.execute(
        """
        INSERT INTO patients (nom, age, diagnostic)
        VALUES (?, ?, ?)
        """,
        (f"{nom}-{sexe}", age, diag)
    )

conn.commit()
conn.close()

print("129 patients ORL ajoutés ✔")
```

```python id="ml4"
import mariadb
import pandas as pd

from sklearn.preprocessing import LabelEncoder
from sklearn.tree import DecisionTreeClassifier

# =========================
# CONNEXION DB
# =========================
conn = mariadb.connect(
    host="localhost",
    user="root",
    password="root123",
    database="orl_ia"
)

# =========================
# LECTURE TABLE
# =========================
query = "SELECT nom, age, diagnostic FROM patients"

df = pd.read_sql(query, conn)

conn.close()

# =========================
# EXTRACTION SEXE
# =========================
df["sexe"] = df["nom"].apply(
    lambda x: 1 if "-M" in x else 0
)

# =========================
# FEATURES
# =========================
X = df[["age", "sexe"]]

# =========================
# TARGET
# =========================
encoder = LabelEncoder()

y = encoder.fit_transform(df["diagnostic"])

# =========================
# MODEL
# =========================
model = DecisionTreeClassifier()

model.fit(X, y)

# =========================
# TEST PREDICTION
# =========================
prediction = model.predict([[60, 1]])

diag = encoder.inverse_transform(prediction)

print("Prediction :", diag[0])
```

```python
from fastapi import FastAPI
from pydantic import BaseModel

import mariadb
import pandas as pd

from sklearn.preprocessing import LabelEncoder
from sklearn.tree import DecisionTreeClassifier

app = FastAPI(title="ORL IA API")

# =========================
# CONNEXION DATABASE
# =========================
def get_connection():
    return mariadb.connect(
        host="localhost",
        user="root",
        password="root123",
        database="orl_ia"
    )

# =========================
# MODELES
# =========================
class Patient(BaseModel):
    nom: str
    age: int
    diagnostic: str

class PredictInput(BaseModel):
    age: int
    sexe: int

# =========================
# TRAIN IA MODEL
# =========================
def train_model():

    conn = get_connection()

    cur = conn.cursor()

    cur.execute(
        "SELECT nom, age, diagnostic FROM patients"
    )

    rows = cur.fetchall()

    df = pd.DataFrame(
        rows,
        columns=["nom", "age", "diagnostic"]
    )

    conn.close()

    # extraction sexe
    df["sexe"] = df["nom"].apply(
        lambda x: 1 if "-M" in x else 0
    )

    # features
    X = df[["age", "sexe"]]

    # target
    encoder = LabelEncoder()

    y = encoder.fit_transform(df["diagnostic"])

    # model
    model = DecisionTreeClassifier()

    model.fit(X, y)

    return model, encoder

# =========================
# LOAD MODEL
# =========================
model, encoder = train_model()

# =========================
# HOME
# =========================
@app.get("/")
def home():
    return {"message": "API ORL IA active"}

# =========================
# GET PATIENTS
# =========================
@app.get("/patients")
def get_patients():

    conn = get_connection()

    cur = conn.cursor()

    cur.execute("SELECT * FROM patients")

    rows = cur.fetchall()

    conn.close()

    return [
        {
            "id": r[0],
            "nom": r[1],
            "age": r[2],
            "diagnostic": r[3]
        }
        for r in rows
    ]

# =========================
# ADD PATIENT
# =========================
@app.post("/patients")
def add_patient(p: Patient):

    conn = get_connection()

    cur = conn.cursor()

    cur.execute(
        """
        INSERT INTO patients (nom, age, diagnostic)
        VALUES (?, ?, ?)
        """,
        (p.nom, p.age, p.diagnostic)
    )

    conn.commit()

    conn.close()

    return {"message": "Patient ajouté ✔"}

# =========================
# STATS
# =========================
@app.get("/stats")
def get_stats():

    conn = get_connection()

    cur = conn.cursor()

    stats = {}

    cur.execute("SELECT COUNT(*) FROM patients")
    stats["total_patients"] = cur.fetchone()[0]

    cur.execute("""
        SELECT COUNT(*)
        FROM patients
        WHERE diagnostic='Sinusite'
    """)
    stats["sinusites"] = cur.fetchone()[0]

    cur.execute("""
        SELECT COUNT(*)
        FROM patients
        WHERE diagnostic='Allergie naso-sinusienne'
    """)
    stats["allergies"] = cur.fetchone()[0]

    cur.execute("""
        SELECT COUNT(*)
        FROM patients
        WHERE diagnostic='Cancer du cavum'
    """)
    stats["cancers_cavum"] = cur.fetchone()[0]

    cur.execute("""
        SELECT COUNT(*)
        FROM patients
        WHERE diagnostic='Cancer sinus maxillaire'
    """)
    stats["cancers_sinus_maxillaire"] = cur.fetchone()[0]

    conn.close()

    return stats

# =========================
# IA PREDICTION
# =========================
@app.post("/predict")
def predict(data: PredictInput):

    sample = pd.DataFrame(
        [[data.age, data.sexe]],
        columns=["age", "sexe"]
    )

    prediction = model.predict(sample)

    diagnostic = encoder.inverse_transform(
        prediction
    )

    return {
        "prediction": diagnostic[0]
    }
```

```python
import random
import mariadb

conn = mariadb.connect(
    host="localhost",
    user="root",

    database="orl_ia"
)

cur = conn.cursor()

# =========================

cur.execute("DELETE FROM patients")

# =========================
# PATHOLOGIES (129)
# =========================
pathologies = (
    ["Cancer du cavum"] * 7 +
    ["Cancer sinus maxillaire"] * 5 +
    ["Sinusite chronique"] * 53 +
    ["Allergie naso-sinusienne"] * 55 +
    ["Autre pathologie ORL"] * 9
)

noms = [

]

def generate_symptoms(diagnostic):
    """
    Logique médicale simplifiée ORL
    """

    if diagnostic == "Sinusite chronique":
        return (1, 1, 0, 0, 0)

    elif diagnostic == "Allergie naso-sinusienne":
        return (1, 1, 0, 0, 0)

    elif diagnostic == "Cancer du cavum":
        return (0, 0, 1, 1, 1)

    elif diagnostic == "Cancer sinus maxillaire":
        return (1, 0, 1, 1, 1)

    else:  # Autre ORL
        return (0, 0, 0, 0, 0)

# =========================
for diag in pathologies:

    nom = random.choice(noms)

    obstruction, rhinorrhee, epistaxis, anosmie, tabac = generate_symptoms(diag)

    sexe = random.choice(["M", "F"])

    cur.execute(
        """
```

```python id="mainapi1"
from fastapi import FastAPI
from pydantic import BaseModel
import mariadb
import pandas as pd

from sklearn.preprocessing import LabelEncoder
from sklearn.tree import DecisionTreeClassifier

app = FastAPI(title="ORL IA API")

# =========================
# CONNEXION DB
# =========================
def get_connection():
    return mariadb.connect(
        host="localhost",
        user="root",
        password="root123",
        database="orl_ia"
    )

# =========================
# CHARGEMENT DATA + TRAIN MODEL
# =========================
def train_model():
    conn = get_connection()
    cur = conn.cursor()

    cur.execute("""
        SELECT age, nom, diagnostic,
               obstruction_nasale,
               rhinorrhee,
               epistaxis,
               anosmie,
               tabac
        FROM patients
    """)

    rows = cur.fetchall()
    conn.close()

    df = pd.DataFrame(
        rows,
        columns=[
            "age", "nom", "diagnostic",
            "obstruction_nasale",
            "rhinorrhee",
            "epistaxis",
            "anosmie",
            "tabac"
        ]
    )

    # sexe
    df["sexe"] = df["nom"].apply(lambda x: 1 if "-M" in x else 0)

    X = df[
        [
            "age", "sexe",
            "obstruction_nasale",
            "rhinorrhee",
            "epistaxis",
            "anosmie",
            "tabac"
        ]
    ]

    encoder = LabelEncoder()
    y = encoder.fit_transform(df["diagnostic"])

    model = DecisionTreeClassifier(max_depth=6)
    model.fit(X, y)

    return model, encoder


model, encoder = train_model()

# =========================
# MODELE API INPUT
# =========================
class PatientInput(BaseModel):
    age: int
    sexe: int
    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int

# =========================
# ROUTE PATIENTS
# =========================
@app.get("/patients")
def get_patients():
    conn = get_connection()
    cur = conn.cursor()

    cur.execute("SELECT * FROM patients")
    rows = cur.fetchall()

    conn.close()

    return rows

# =========================
# ROUTE PREDICTION IA
# =========================
@app.post("/predict")
def predict(p: PatientInput):

    sample = pd.DataFrame(
        [[
            p.age,
            p.sexe,
            p.obstruction_nasale,
            p.rhinorrhee,
            p.epistaxis,
            p.anosmie,
            p.tabac
        ]],
        columns=[
            "age", "sexe",
            "obstruction_nasale",
            "rhinorrhee",
            "epistaxis",
            "anosmie",
            "tabac"
        ]
    )

    pred = model.predict(sample)
    result = encoder.inverse_transform(pred)

    return {
        "prediction": result[0]
    }
```

```python id="rf1"
import mariadb
import pandas as pd
import joblib

from sklearn.preprocessing import LabelEncoder

conn = mariadb.connect(
    host="localhost",

    database="orl_ia"
)

cur = conn.cursor()

cur.execute("""
    SELECT age, nom, diagnostic,
           obstruction_nasale,
           rhinorrhee,
           epistaxis,
           anosmie,
           tabac

df = pd.DataFrame(rows, columns=[

    "epistaxis",
    "anosmie",
```

```python id="finalmain1"
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel, Field
import mariadb
import joblib

app = FastAPI(title="ORL IA API")

# =========================
# MODELE
# =========================
model = joblib.load("model_orl.pkl")
encoder = joblib.load("encoder_orl.pkl")

# =========================
# DB CONNECTION
# =========================
def get_connection():
    try:
        return mariadb.connect(
            host="localhost",
            user="root",
            password="root123",
            database="orl_ia"
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

# =========================
# INPUT VALIDÉ
# =========================
class PatientInput(BaseModel):
    age: int = Field(ge=0, le=120)
    sexe: int = Field(ge=0, le=1)
    obstruction_nasale: int = Field(ge=0, le=1)
    rhinorrhee: int = Field(ge=0, le=1)
    epistaxis: int = Field(ge=0, le=1)
    anosmie: int = Field(ge=0, le=1)
    tabac: int = Field(ge=0, le=1)

# =========================
# PATIENTS
# =========================
@app.get("/patients")
def get_patients():
    conn = get_connection()
    cur = conn.cursor()

    try:
        cur.execute("SELECT * FROM patients")
        rows = cur.fetchall()
        return rows
    finally:
        conn.close()

# =========================
# PREDICTION IA
# =========================
@app.post("/predict")
def predict(p: PatientInput):

    sample = [[
        p.age,
        p.sexe,
        p.obstruction_nasale,
        p.rhinorrhee,
        p.epistaxis,
        p.anosmie,
        p.tabac
    ]]

    pred = model.predict(sample)
    result = encoder.inverse_transform(pred)

    return {
        "prediction": result[0]
    }
```

# 🧠 MAIN.PY — VERSION DOCKER READY

```python id="docker_main_full"
from fastapi import FastAPI
from pydantic import BaseModel
import mariadb
import joblib
import os

app = FastAPI(title="ORL IA API - DOCKER READY")

# =========================
# MODELE IA
# =========================
model = joblib.load("model_orl.pkl")
encoder = joblib.load("encoder_orl.pkl")

# =========================
# CONNEXION DB (DOCKER FIX)
# =========================
def get_connection():
    return mariadb.connect(
        host=os.getenv("DB_HOST", "localhost"),
        user=os.getenv("DB_USER", "root"),
        password=os.getenv("DB_PASSWORD", "root123"),
        database=os.getenv("DB_NAME", "orl_ia")
    )

# =========================
# INPUT PATIENT
# =========================
class PatientInput(BaseModel):
    age: int
    sexe: int
    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int

# =========================
# ROUTE - LIST PATIENTS
# =========================
@app.get("/patients")
def get_patients():
    conn = get_connection()
    cur = conn.cursor()

    try:
        cur.execute("SELECT * FROM patients")
        rows = cur.fetchall()
        return rows
    finally:
        conn.close()

# =========================
# ROUTE - PREDICTION IA
# =========================
@app.post("/predict")
def predict(p: PatientInput):

    sample = [[
        p.age,
        p.sexe,
        p.obstruction_nasale,
        p.rhinorrhee,
        p.epistaxis,
        p.anosmie,
        p.tabac
    ]]

    pred = model.predict(sample)
    result = encoder.inverse_transform(pred)

    return {
        "prediction": result[0]
    }

# =========================
# ROUTE - RISK SCORE
# =========================
@app.post("/risk")
def risk(p: PatientInput):

    score = 0

    if p.tabac == 1:
        score += 40
    if p.epistaxis == 1:
        score += 25
    if p.anosmie == 1:
        score += 15
    if p.obstruction_nasale == 1:
        score += 10
    if p.rhinorrhee == 1:
        score += 5
    if p.age > 50:
        score += 10

    if score > 100:
        score = 100

    if score < 30:
        niveau = "Faible"
    elif score < 60:
        niveau = "Modéré"
    else:
        niveau = "Élevé"

    return {
        "score": score,
        "niveau_risque": niveau
    }
```

# 🧠 5. main.py (FULL VERSION FINAL)

```python
from fastapi import FastAPI
from pydantic import BaseModel
import mariadb
import joblib
import pandas as pd
import os
from dotenv import load_dotenv

load_dotenv()

app = FastAPI(title="ORL IA API DOCKER")

# =========================
# MODELE ML
# =========================
model = joblib.load("model_orl.pkl")
encoder = joblib.load("encoder_orl.pkl")

# =========================
# CONNEXION DB
# =========================
def get_connection():
    return mariadb.connect(
        host=os.getenv("DB_HOST"),
        port=int(os.getenv("DB_PORT")),
        user=os.getenv("DB_USER"),
        password=os.getenv("DB_PASSWORD"),
        database=os.getenv("DB_NAME")
    )

# =========================
# INPUT API
# =========================
class PatientInput(BaseModel):
    age: int
    sexe: int
    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int

# =========================
# GET PATIENTS
# =========================
@app.get("/patients")
def get_patients():
    conn = get_connection()
    cur = conn.cursor()

    cur.execute("SELECT * FROM patients")
    rows = cur.fetchall()

    conn.close()
    return rows

# =========================
# PREDICTION
# =========================
@app.post("/predict")
def predict(p: PatientInput):

    sample = [[
        p.age,
        p.sexe,
        p.obstruction_nasale,
        p.rhinorrhee,
        p.epistaxis,
        p.anosmie,
        p.tabac
    ]]

    pred = model.predict(sample)
    result = encoder.inverse_transform(pred)

    return {"prediction": result[0]}

# =========================
# SCORE RISQUE
# =========================
@app.post("/risk")
def risk(p: PatientInput):

    score = (
        p.obstruction_nasale +
        p.rhinorrhee +
        p.epistaxis +
        p.anosmie +
        p.tabac
    ) * 20

    niveau = "Faible"
    if score > 60:
        niveau = "Élevé"
    elif score > 30:
        niveau = "Modéré"

    return {
        "score": score,
        "niveau_risque": niveau
    }
```

```python
import mariadb
import random

conn = mariadb.connect(
    host="localhost",
    port=3307,
    user="root",
    password="root123",
    database="orl_ia"
)

cur = conn.cursor()

cur.execute("DELETE FROM patients")

pathologies = [
    ("Cancer du cavum", 7),
    ("Cancer du sinus maxillaire", 5),
    ("Sinusite chronique", 53),
    ("Allergie naso-sinusienne", 55),
    ("Autre pathologie rhinologique", 9)
]

def generate_sex():
    return random.choice(["-M", "-F"])

def generate_age():
    return random.randint(5, 85)

data = []

for diag, count in pathologies:
    for _ in range(count):
        nom = f"Patient{random.randint(1000,9999)}{generate_sex()}"
        age = generate_age()

        data.append((nom, age, diag, 0, 0, 0, 0, 0))

cur.executemany("""
INSERT INTO patients
(nom, age, diagnostic,
obstruction_nasale, rhinorrhee, epistaxis, anosmie, tabac)
VALUES (?, ?, ?, ?, ?, ?, ?, ?)
""", data)

conn.commit()
conn.close()

print("129 patients ORL générés ✔")
```

```python
import mariadb
import pandas as pd
import joblib
from sklearn.tree import DecisionTreeClassifier
from sklearn.preprocessing import LabelEncoder

conn = mariadb.connect(
    host="localhost",
    port=3307,
    user="root",
    password="root123",
    database="orl_ia"
)

df = pd.read_sql("SELECT * FROM patients", conn)
conn.close()

df["sexe"] = df["nom"].apply(lambda x: 1 if "-M" in x else 0)

features = [
    "age","sexe",
    "obstruction_nasale",
    "rhinorrhee",
    "epistaxis",
    "anosmie",
    "tabac"
]

X = df[features]

encoder = LabelEncoder()
y = encoder.fit_transform(df["diagnostic"])

model = DecisionTreeClassifier(max_depth=6)
model.fit(X, y)

joblib.dump(model, "model_orl.pkl")
joblib.dump(encoder, "encoder_orl.pkl")

print("Model trained ✔")
```

```python
from fastapi import FastAPI
from pydantic import BaseModel
import mariadb
import joblib
import pandas as pd
import os

app = FastAPI(title="ORL IA API")

# =========================
# MODELE
# =========================
model = joblib.load("model_orl.pkl")
encoder = joblib.load("encoder_orl.pkl")

# =========================
# DB CONNECTION (DOCKER SAFE)
# =========================
def get_connection():
    return mariadb.connect(
        host=os.getenv("DB_HOST", "127.0.0.1"),
        port=int(os.getenv("DB_PORT", 3307)),
        user=os.getenv("DB_USER", "root"),
        password=os.getenv("DB_PASSWORD", "root123"),
        database=os.getenv("DB_NAME", "orl_ia")
    )

# =========================
# INPUT
# =========================
class PatientInput(BaseModel):
    age: int
    sexe: int
    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int

# =========================
# GET PATIENTS
# =========================
@app.get("/patients")
def get_patients():
    conn = get_connection()
    cur = conn.cursor()
    cur.execute("SELECT * FROM patients")
    rows = cur.fetchall()
    conn.close()
    return rows

# =========================
# PREDICTION
# =========================
@app.post("/predict")
def predict(p: PatientInput):

    sample = [[
        p.age,
        p.sexe,
        p.obstruction_nasale,
        p.rhinorrhee,
        p.epistaxis,
        p.anosmie,
        p.tabac
    ]]

    pred = model.predict(sample)
    result = encoder.inverse_transform(pred)

    return {"prediction": result[0]}


# =========================
# RISK SIMPLE
# =========================
@app.post("/risk")
def risk(p: PatientInput):

    score = (
        p.age * 0.3 +
        p.obstruction_nasale * 15 +
        p.rhinorrhee * 10 +
        p.epistaxis * 20 +
        p.anosmie * 25 +
        p.tabac * 30
    )

    if score >= 70:
        niveau = "Élevé"
    elif score >= 40:
        niveau = "Moyen"
    else:
        niveau = "Faible"

    return {
        "score": round(score),
        "niveau_risque": niveau
    }
```

Code COMPLET :

```python
from fastapi import FastAPI
from pydantic import BaseModel

import mariadb
import pandas as pd
import joblib
import os

app = FastAPI(title="ORL IA API DOCKER")

# =========================
# LOAD MODEL
# =========================
model = joblib.load("model_orl.pkl")
encoder = joblib.load("encoder_orl.pkl")

# =========================
# DB CONNECTION
# =========================
def get_connection():

    return mariadb.connect(
        host=os.getenv("DB_HOST", "mariadb"),
        port=int(os.getenv("DB_PORT", 3306)),
        user=os.getenv("DB_USER", "root"),
        password=os.getenv("DB_PASSWORD", "root123"),
        database=os.getenv("DB_NAME", "orl_ia")
    )

# =========================
# INPUT MODEL
# =========================
class PatientInput(BaseModel):

    age: int
    sexe: int

    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int

# =========================
# HOME
# =========================
@app.get("/")
def home():

    return {
        "message": "ORL IA API DOCKER OK"
    }

# =========================
# GET PATIENTS
# =========================
@app.get("/patients")
def get_patients():

    conn = get_connection()

    cur = conn.cursor()

    cur.execute("SELECT * FROM patients")

    rows = cur.fetchall()

    conn.close()

    return rows

# =========================
# PREDICT
# =========================
@app.post("/predict")
def predict(p: PatientInput):

    sample = pd.DataFrame(
        [[
            p.age,
            p.sexe,
            p.obstruction_nasale,
            p.rhinorrhee,
            p.epistaxis,
            p.anosmie,
            p.tabac
        ]],
        columns=[
            "age",
            "sexe",
            "obstruction_nasale",
            "rhinorrhee",
            "epistaxis",
            "anosmie",
            "tabac"
        ]
    )

    pred = model.predict(sample)

    result = encoder.inverse_transform(pred)

    return {
        "prediction": result[0]
    }
```

# 🐍 5. API FASTAPI (PRODUCTION main.py)

📁 `api/main.py`

```python
from fastapi import FastAPI
from pydantic import BaseModel
import mariadb
import joblib
import pandas as pd
import os

app = FastAPI(title="ORL IA FULL STACK")

model = joblib.load("model_orl.pkl")
encoder = joblib.load("encoder_orl.pkl")


def get_connection():
    return mariadb.connect(
        host=os.getenv("DB_HOST"),
        port=int(os.getenv("DB_PORT")),
        user=os.getenv("DB_USER"),
        password=os.getenv("DB_PASSWORD"),
        database=os.getenv("DB_NAME")
    )


class PatientInput(BaseModel):
    age: int
    sexe: int
    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int


@app.get("/patients")
def get_patients():
    conn = get_connection()
    cur = conn.cursor()
    cur.execute("SELECT * FROM patients")
    data = cur.fetchall()
    conn.close()
    return data


@app.post("/predict")
def predict(p: PatientInput):

    X = [[
        p.age,
        p.sexe,
        p.obstruction_nasale,
        p.rhinorrhee,
        p.epistaxis,
        p.anosmie,
        p.tabac
    ]]

    pred = model.predict(X)
    return {"prediction": encoder.inverse_transform(pred)[0]}


@app.post("/risk")
def risk(p: PatientInput):

    score = (
        p.age * 0.3 +
        p.obstruction_nasale * 15 +
        p.rhinorrhee * 10 +
        p.epistaxis * 20 +
        p.anosmie * 25 +
        p.tabac * 30
    )

    return {
        "score": int(score),
        "niveau": "Élevé" if score > 70 else "Moyen" if score > 40 else "Faible"
    }
```
