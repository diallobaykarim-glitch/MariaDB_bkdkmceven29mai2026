# 🧠 3. API ORL (api_orl/main.py)

👉 tu gardes ton code actuel MAIS version Docker-safe :

```python id="api_orl_final"
import os
from fastapi import FastAPI
from pydantic import BaseModel
import mariadb
import joblib
import pandas as pd

app = FastAPI(title="ORL IA API")

model = joblib.load("model_orl.pkl")
encoder = joblib.load("encoder_orl.pkl")

def get_connection():
    return mariadb.connect(
        host=os.getenv("DB_HOST"),
        port=int(os.getenv("DB_PORT")),
        user=os.getenv("DB_USER"),
        password=os.getenv("DB_PASSWORD"),
        database=os.getenv("DB_NAME")
    )

class PatientInput(BaseModel):
    age: int
    sexe: int
    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int

@app.get("/patients")
def patients():
    conn = get_connection()
    cur = conn.cursor()
    cur.execute("SELECT * FROM patients")
    data = cur.fetchall()
    conn.close()
    return data

@app.post("/predict")
def predict(p: PatientInput):
    df = pd.DataFrame([[p.age,p.sexe,p.obstruction_nasale,p.rhinorrhee,
                        p.epistaxis,p.anosmie,p.tabac]])

    pred = model.predict(df)
    return {"prediction": encoder.inverse_transform(pred)[0]}
```

```python id="oracle_api"
from fastapi import FastAPI
import oracledb

app = FastAPI(title="ORACLE IA API")

def get_conn():
    return oracledb.connect(
        user="system",
        password="oracle",
        dsn="oracle-db:1521/XEPDB1"
    )

@app.get("/patients")
def patients():
    conn = get_conn()
    cur = conn.cursor()
    cur.execute("SELECT * FROM patients")
    rows = cur.fetchall()
    conn.close()
    return rows
```

```python
import os
import mariadb
from fastapi import FastAPI
from pydantic import BaseModel
import joblib

app = FastAPI()

model = joblib.load("model_orl.pkl")

def get_connection():
    return mariadb.connect(
        host=os.getenv("DB_HOST", "localhost"),
        port=int(os.getenv("DB_PORT", 3306)),
        user=os.getenv("DB_USER", "root"),
        password=os.getenv("DB_PASSWORD", "root123"),
        database=os.getenv("DB_NAME", "orl_ia")
    )

class Patient(BaseModel):
    age: int
    sexe: int
    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int

@app.get("/patients")
def get_patients():
    conn = get_connection()
    cur = conn.cursor()
    cur.execute("SELECT * FROM patients")
    rows = cur.fetchall()
    conn.close()
    return rows

@app.post("/predict")
def predict(p: Patient):
    X = [[p.age, p.sexe, p.obstruction_nasale, p.rhinorrhee,
          p.epistaxis, p.anosmie, p.tabac]]
    pred = model.predict(X)[0]
    return {"prediction": str(pred)}
```

```python
import os
import joblib
from fastapi import FastAPI
from pydantic import BaseModel
from api.db import get_connection

app = FastAPI(title="ORL IA API PRO")

model = joblib.load("model_orl.pkl")

class Patient(BaseModel):
    age: int
    sexe: int
    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int


@app.get("/")
def home():
    return {"status": "OK", "message": "ORL IA API running"}


@app.get("/patients")
def get_patients():
    conn = get_connection()
    cur = conn.cursor()
    cur.execute("SELECT * FROM patients")
    data = cur.fetchall()
    conn.close()
    return data


@app.post("/predict")
def predict(p: Patient):
    X = [[p.age, p.sexe, p.obstruction_nasale,
          p.rhinorrhee, p.epistaxis, p.anosmie, p.tabac]]
    
    pred = model.predict(X)[0]
    return {"prediction": str(pred)}


@app.post("/risk")
def risk(p: Patient):
    score = (p.age * 0.5 +
             p.epistaxis * 20 +
             p.obstruction_nasale * 15 +
             p.anosmie * 15)

    niveau = "Élevé" if score > 60 else "Modéré" if score > 30 else "Faible"

    return {"score": score, "niveau_risque": niveau}
```

Le seul problème maintenant est **un conflit de conteneur MariaDB** :

> `/orl_mariadb` existe déjà, donc Docker refuse d’en recréer un nouveau.

---

## 🔴 Cause

Un ancien conteneur MariaDB tourne encore (ou est arrêté mais pas supprimé) avec le même nom :

```
orl_mariadb
```

---

## ✅ Solution simple (recommandée)

### 1. Voir le conteneur existant

```powershell
docker ps -a
```

Cherche :

```
orl_mariadb
```

---

### 2. Le supprimer

```powershell
docker rm -f orl_mariadb
```

---

### 3. Relancer ton projet

```powershell
docker-compose up
```

---

## 🧹 Solution “reset total” (si tu veux repartir propre)

```powershell
docker-compose down -v
docker rm -f orl_mariadb
docker system prune -f
docker-compose up
```

---

## ⚠️ Pourquoi `docker-compose down` n’a pas suffi ?

Ça arrive quand :

* le conteneur a été créé **hors compose**
* ou renommé manuellement
* ou un ancien projet Docker utilise encore `orl_mariadb`

---

## 💡 Astuce pour éviter ça à l’avenir

Dans ton `docker-compose.yml`, utilise un nom de projet isolé :

```bash
docker-compose -p maria_project up
```

ou change le nom du service/container pour éviter les collisions.

---

Si tu veux, envoie ton `docker-compose.yml` et je peux te le sécuriser pour éviter ce type de conflit définitivement.

## ⚠️ Point important (tu as un environnement “multi-DB”)

Tu as en réalité **3 systèmes qui traînent en parallèle** :

* MariaDB → `orl_mariadb`
* Oracle → `oracle_db`
* 3 APIs différentes :

  * `api_orl`
  * `api_oracle`
  * `dashboard_global`

👉 Ça veut dire que tu as probablement **plusieurs anciens docker-compose ou projets qui se chevauchent**.

---

## 🧹 Recommandation propre (très importante)

Si tu veux éviter ce genre de conflit définitivement :

### 1. Stop tout

```powershell
docker stop $(docker ps -aq)
```

### 2. Suppression globale (clean reset)

```powershell
docker rm -f $(docker ps -aq)
```

### 3. Nettoyage volumes

```powershell
docker volume prune -f
```

### 4. Relancer uniquement TON projet actuel

```powershell
docker-compose up --build
```

---

* utilises **un seul docker-compose actif**
* supprimes les anciens projets ORL / Oracle inutilisés
* gardes uniquement :

  * Streamlit dashboard

Ton problème est **clair et unique** ici :

> ❌ `FileNotFoundError: model_orl.pkl`

Donc ton conteneur `orl_api` plante parce que **le fichier du modèle n’existe pas dans l’image Docker**.

---

# 🧠 Pourquoi ça arrive

Dans ton `main.py` tu as :

```python
model = joblib.load("model_orl.pkl")
```

Mais dans ton Dockerfile API :

```dockerfile
COPY . .
```

👉 Donc Docker copie seulement ce qui est dans `api/`

Or ton modèle :

* ❌ n’est pas dans `api/`
* ❌ n’est pas copié dans l’image
* ❌ donc Uvicorn crash au démarrage

---

Structure actuelle probable :

```
MariaDB_project/
```

👉 MAIS Docker build ne le voit pas.

```dockerfile
COPY main.py .
```

# ✅ main.py (version actualisée)

```python
from fastapi import FastAPI
from pydantic import BaseModel
import joblib
import os

app = FastAPI(
    title="ORL IA API PRO",
    version="0.1.0"
)

# ----------------------------
# CHARGEMENT DU MODÈLE
# ----------------------------
MODEL_PATH = os.path.join(os.path.dirname(__file__), "model_orl.pkl")
model = joblib.load(MODEL_PATH)

# ----------------------------
# LABELS MÉDICAUX
# ----------------------------
labels = {
    0: "Rhinite chronique",
    1: "Allergie naso-sinusienne",
    2: "Sinusite chronique",
    3: "Cancer ethmoïde"
}

# ----------------------------
# SCHEMA PREDICTION
# ----------------------------
class PatientInput(BaseModel):
    age: int
    sexe: int
    obstruction_nasale: int
    rhinorrhee: int
    epistaxis: int
    anosmie: int
    tabac: int


# ----------------------------
# HOME
# ----------------------------
@app.get("/")
def home():
    return {
        "status": "OK",
        "message": "ORL IA API running"
    }


# ----------------------------
# GET PATIENTS
# ----------------------------
@app.get("/patients")
def get_patients():
    import mysql.connector

    conn = mysql.connector.connect(
        host="mariadb",
        user="root",
        password="root123",
        database="orl_ia"
    )

    cursor = conn.cursor()
    cursor.execute("SELECT * FROM patients")
    rows = cursor.fetchall()

    conn.close()
    return rows


# ----------------------------
# PREDICTION ML
# ----------------------------
@app.post("/predict")
def predict(patient: PatientInput):

    data = [[
        patient.age,
        patient.sexe,
        patient.obstruction_nasale,
        patient.rhinorrhee,
        patient.epistaxis,
        patient.anosmie,
        patient.tabac
    ]]

    prediction = model.predict(data)[0]

    return {
        "prediction_id": int(prediction),
        "prediction_label": labels.get(int(prediction), "Inconnu")
    }


# ----------------------------
# RISK SCORE (simple logique métier)
# ----------------------------
@app.post("/risk")
def risk(patient: PatientInput):

    score = (
        patient.age * 0.02 +
        patient.tabac * 2 +
        patient.epistaxis * 1.5 +
        patient.anosmie * 1.2
    )

    if score < 2:
        level = "LOW"
    elif score < 5:
        level = "MEDIUM"
    else:
        level = "HIGH"

    return {
        "risk_score": round(score, 2),
        "risk_level": level
    }
```

```python
    epistaxis: int
    anosmie: int
    tabac: int

        "status": "OK",
        "message": "ORL IA API running"
    }

@app.get("/patients")
def get_patients():
    try:
        conn = get_connection()
        cursor = conn.cursor()

        cursor.execute("SELECT * FROM patients")
        columns = [desc[0] for desc in cursor.description]
        rows = cursor.fetchall()

        result = [dict(zip(columns, row)) for row in rows]

        cursor.close()
        conn.close()

        return {
            "count": len(result),
            "data": result
        }

    except Exception as e:
        return {"error": str(e)}


# =========================
# PREDICTION (IA SIMULÉE)
# =========================
@app.post("/predict")
def predict(patient: PatientInput):
    try:
        score = (
            patient.obstruction_nasale +
            patient.rhinorrhee +
            patient.epistaxis +
            patient.tabac +
            (1 if patient.age > 50 else 0)
        )

        if score >= 4:
            prediction_id = 3
            label = "Cancer ethmoïde"
        elif score == 3:
            prediction_id = 1
            label = "Cancer sinus maxillaire"
        elif score == 2:
            prediction_id = 2
            label = "Allergie naso-sinusienne"
        else:
            prediction_id = 4
            label = "Rhinite chronique"

        return {
            "prediction_id": prediction_id,
            "prediction_label": label
        }

    except Exception as e:
        return {"error": str(e)}


# =========================
# RISK SCORE
# =========================
@app.post("/risk")
def risk(patient: PatientInput):
    try:
        risk_score = (
            patient.age / 20 +
            patient.obstruction_nasale * 1.5 +
            patient.rhinorrhee * 1.2 +
            patient.epistaxis * 1.8 +
            patient.tabac * 1.5
        )

        if risk_score >= 6:
            level = "HIGH"
        elif risk_score >= 3:
            level = "MEDIUM"
        else:
            level = "LOW"

        return {
            "risk_score": round(risk_score, 2),
            "risk_level": level
        }

    except Exception as e:
        return {"error": str(e)}
```
