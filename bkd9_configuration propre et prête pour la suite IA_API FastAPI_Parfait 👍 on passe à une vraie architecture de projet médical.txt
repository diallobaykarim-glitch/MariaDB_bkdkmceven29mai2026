```python
import random
import mariadb

conn = mariadb.connect(
    host="localhost",
    user="root",
    password="root123",
    database="orl_ia"
)

cur = conn.cursor()

# reset table
cur.execute("DELETE FROM patients")

pathologies = (
    ["Cancer du cavum"] * 7 +
    ["Cancer sinus maxillaire"] * 5 +
    ["Sinusite"] * 53 +
    ["Allergie naso-sinusienne"] * 55 +
    ["Autre pathologie ORL"] * 9
)

noms = ["Diallo", "Sow", "Ndiaye", "Ba", "Fall", "Diop", "Sarr", "Ndao", "Kane", "Gueye"]

for i in range(129):
    nom = random.choice(noms)
    age = random.randint(5, 85)
    sexe = random.choice(["M", "F"])
    diag = pathologies[i]

    cur.execute(
        "INSERT INTO patients (nom, age, diagnostic) VALUES (?, ?, ?)",
        (f"{nom}-{sexe}", age, diag)
    )

conn.commit()
conn.close()

print("129 patients ORL insérés ✔")
```
